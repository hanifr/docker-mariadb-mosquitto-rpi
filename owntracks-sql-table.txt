CREATE TABLE tracker (
tid VARCHAR(150) NOT NULL,
bat DECIMAL(16,0) DEFAULT '0.00' NOT NULL,
vel DECIMAL(16,0) DEFAULT '0.00' NOT NULL,
alt DECIMAL(16,0) DEFAULT '0.00' NOT NULL,
lat DECIMAL(16,6) DEFAULT '0.00' NOT NULL,
lon DECIMAL(16,6) DEFAULT '0.00' NOT NULL,
timestamp TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
